sql = statement_or_query eof;
statement_or_query =
  | statement
  | query;
statement = 
  | CREATE => create
  | INSERT => insert;
create =
  | (CREATE TABLE) => create_table;
create_table = CREATE TABLE name open_paren column_specs close_paren;
insert = INSERT INTO name open_paren column_specs close_paren table_expr;
table_expr = 
  | VALUES => table_expr_values;
table_expr_values = VALUES open_paren value_exprs close_paren;
value_exprs = value_expr*comma;
value_expr =
  | value;
value =
  | number
  | string;
column_specs = column_spec*comma;
column_spec = name typ?;
typ = name;

tokens =
  | less_than
  | greater_than
  | less_than_or_equal
  | greater_than_or_equal
  | not_less_than
  | not_greater_than
  | equal
  | double_equal
  | not_equal
  | plus
  | minus
  | star
  | forward_slash
  | dot
  | modulus
  | bitwise_and
  | bitwise_or
  | bitwise_not
  | string_concat
  | shift_left
  | shift_right
;
